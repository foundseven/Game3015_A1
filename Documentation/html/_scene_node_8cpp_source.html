202XMFLOAT4X4 SceneNode::getWorldTransform() const
203{
204 XMFLOAT4X4 transform = MathHelper::Identity4x4();
205 XMMATRIX T = XMLoadFloat4x4(&transform);
206
207 for (const SceneNode* node = this; node != nullptr; node = node->mParent)
208 {
209 auto nodeTransform = node->getTransform();
210 XMMATRIX Tp = XMLoadFloat4x4(&nodeTransform);
211 T = Tp * T;
212 }
213 XMStoreFloat4x4(&transform, T);
214
215 return transform;
216}

222XMFLOAT4X4 SceneNode::getTransform() const
223{
224 XMFLOAT4X4 transform;
225 XMStoreFloat4x4(&transform, XMMatrixScaling(mWorldScaling.x, mWorldScaling.y, mWorldScaling.z) *

227 XMMatrixRotationY(mWorldRotation.y) *

229 XMMatrixTranslation(mWorldPosition.x, mWorldPosition.y, mWorldPosition.z));
