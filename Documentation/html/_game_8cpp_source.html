46bool Game::Initialize()
47{
48 if (!D3DApp::Initialize())
49 return false;
50
51 // Set initial camera position and orientation
52 mCamera.SetPosition(0, 8, 0);
53 mCamera.Pitch(3.14 / 2);
54
55 // Reset the command list to prep for initialization commands.
56 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));
57
58 // Get the increment size of a descriptor in this heap type
59 mCbvSrvDescriptorSize = md3dDevice->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_CBV_SRV_UAV);
60
61 // Build the core game resources
62 LoadTextures();
63 BuildRootSignature();
64 BuildDescriptorHeaps();
65 BuildShadersAndInputLayout();
66 BuildShapeGeometry();
67 BuildHillGeometry();
68 BuildMaterials();
69 RegisterStates();
70 mStateStack.pushState(States::Title);
71
72 BuildPSOs();
73
74 // Execute the initialization commands.
75 ThrowIfFailed(mCommandList->Close());
76 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
77 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
78
79 // Wait until initialization is complete.
80 FlushCommandQueue();
81
82 return true;
83}

112void Game::Update(const GameTimer& gt)
113{
114 // Handle input and game world updates
115 //ProcessInput();
116 //mWorld.update(gt);
117 mStateStack.Update(gt);
118 mStateStack.handleRealTimeInput();
119
120 if (mStateStack.isEmpty())
121 {
122 PostQuitMessage(0);
123 return;
124 }
125
126 UpdateCamera(gt);
127
128 // Cycle through the circular frame resource array.
129 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
130 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
131
132 // Wait for GPU completion
133 if (mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
134 {
135 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
136 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
137 WaitForSingleObject(eventHandle, INFINITE);
138 CloseHandle(eventHandle);
139 }
140
141 // Update scene-dependent constant buffers
142 AnimateMaterials(gt);
143 UpdateObjectCBs(gt);
144 UpdateMaterialCBs(gt);
145 UpdateMainPassCB(gt);
146}

162void Game::Draw(const GameTimer& gt)
163{
164 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
165
166 // Reuse the memory associated with command recording.
167 // We can only reset when the associated command lists have finished execution on the GPU.
168 ThrowIfFailed(cmdListAlloc->Reset());
169
170 // Reuse the command list
171 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
172 // Reusing the command list reuses memory.
173 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mOpaquePSO.Get()));
174
175 // Set the viewport and scissor rectangle
176 mCommandList->RSSetViewports(1, &mScreenViewport);
177 mCommandList->RSSetScissorRects(1, &mScissorRect);
178
179 // Indicate a state transition on the resource usage.
180 auto transition1 = CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
181 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET);
182 mCommandList->ResourceBarrier(1, &transition1);
183
184 // Clear the back buffer and depth buffer.
185 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
186 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
187
188 // Specify render targets
189 auto dsv = DepthStencilView();
190
191 // Specify the buffers we are going to render to.
192 auto buffer = CurrentBackBufferView();
193 mCommandList->OMSetRenderTargets(1, &buffer, true, &dsv);
194
195 // Set descriptor heaps
196 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
197 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
198
199 // Set root signature
200 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
201
202 // Set pass constant buffer
203 auto passCB = mCurrFrameResource->PassCB->Resource();
204 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
205
206 mStateStack.Draw();
207
208 // Transition the back buffer to the present state
209 auto transition2 = CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
210 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT);
211 mCommandList->ResourceBarrier(1, &transition2);
212
213 // Done recording commands.
214 ThrowIfFailed(mCommandList->Close());
215
216 // Add the command list to the queue for execution.
217 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
218 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
219
220 // Present the back buffer
221 ThrowIfFailed(mSwapChain->Present(0, 0));
222 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
223
224 // Advance the fence value to mark commands up to this fence point.
225 mCurrFrameResource->Fence = ++mCurrentFence;
226
227 // Add an instruction to the command queue to set a new fence point.
228 // Because we are on the GPU timeline, the new fence point won't be
229 // set until the GPU finishes processing all the commands prior to this Signal().
230 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
231
232}

300void Game::UpdateObjectCBs(const GameTimer& gt)
301{
302 State* currentState = mStateStack.GetCurrentState();
303
304 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
305 for (auto& e : currentState->getRenderItems())
306 {
307 // Only update the cbuffer data if the constants have changed.
308 if (e->NumFramesDirty > 0)
309 {
310 XMMATRIX world = XMLoadFloat4x4(&e->World);
311 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
312
313 ObjectConstants objConstants;
314 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
315 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
316
317 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
318
319 // Next FrameResource need to be updated too.
320 e->NumFramesDirty--;
321 }
322 }

367void Game::UpdateMainPassCB(const GameTimer& gt)
368{
369 // Get the view and projection matrices from the camera
370 XMMATRIX view = mCamera.GetView();
371 XMMATRIX proj = mCamera.GetProj();
372
373 // Calculate the view-projection matrix
374 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
375
376 // Calculate the inverse view, projection, and view-projection matrices
377 auto detView = XMMatrixDeterminant(view);
378 auto detProj = XMMatrixDeterminant(proj);
379 auto detViewProj = XMMatrixDeterminant(viewProj);
380
381 XMMATRIX invView = XMMatrixInverse(&detView, view);
382 XMMATRIX invProj = XMMatrixInverse(&detProj, proj);
383 XMMATRIX invViewProj = XMMatrixInverse(&detViewProj, viewProj);
384
385 // Store the matrices in the main pass constant buffer
386 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
387 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
388 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
389 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
390 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
391 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
392
393 // Set other pass constants
394 mMainPassCB.EyePosW = mCamera.GetPosition3f();
395 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
396 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
397 mMainPassCB.NearZ = 1.0f;
398 mMainPassCB.FarZ = 1000.0f;
399 mMainPassCB.TotalTime = gt.TotalTime();
400 mMainPassCB.DeltaTime = gt.DeltaTime();
401 mMainPassCB.AmbientLight = { 0.25f, 0.25f, 0.35f, 1.0f };
402 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
403 mMainPassCB.Lights[0].Strength = { 0.6f, 0.6f, 0.6f };
404 mMainPassCB.Lights[1].Direction = { -0.57735f, -0.57735f, 0.57735f };
405 mMainPassCB.Lights[1].Strength = { 0.3f, 0.3f, 0.3f };
406 mMainPassCB.Lights[2].Direction = { 0.0f, -0.707f, -0.707f };
407 mMainPassCB.Lights[2].Strength = { 0.15f, 0.15f, 0.15f };
408
409 // Copy the pass constant buffer data
410 auto currPassCB = mCurrFrameResource->PassCB.get();
411 currPassCB->CopyData(0, mMainPassCB);
412}

473void Game::BuildRootSignature()
474{
475 CD3DX12_DESCRIPTOR_RANGE texTable;
476 texTable.Init(D3D12_DESCRIPTOR_RANGE_TYPE_SRV, 1, 0);
477
478 // Root parameter can be a table, root descriptor or root constants.
479 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
480
481 // Perfomance TIP: Order from most frequent to least frequent.
482 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
483 slotRootParameter[1].InitAsConstantBufferView(0);
484 slotRootParameter[2].InitAsConstantBufferView(1);
485 slotRootParameter[3].InitAsConstantBufferView(2);
486
487 auto staticSamplers = GetStaticSamplers();
488
489 // A root signature is an array of root parameters.
490 //The Init function of the CD3DX12_ROOT_SIGNATURE_DESC class has two parameters that allow you to
491 //define an array of so - called static samplers your application can use.
492 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
493 (UINT)staticSamplers.size(), staticSamplers.data(), //6 samplers!
494 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
495
496 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
497 ComPtr<ID3DBlob> serializedRootSig = nullptr;
498 ComPtr<ID3DBlob> errorBlob = nullptr;
499 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
500 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
501
502 if (errorBlob != nullptr)
503 {
504 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
505 }
506 ThrowIfFailed(hr);
507
508 ThrowIfFailed(md3dDevice->CreateRootSignature(
509 0,
510 serializedRootSig->GetBufferPointer(),
511 serializedRootSig->GetBufferSize(),
512 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
513}

534 //
535 // Fill out the heap with actual descriptors.

538

669void Game::BuildShapeGeometry()
670{
671 //mWorld.buildShapeGeometry(md3dDevice, mCommandList, mGeometries);
672
673 GeometryGenerator geoGen;
674 GeometryGenerator::MeshData box = geoGen.CreateBox(1, 0, 1, 1);
675
676 SubmeshGeometry boxSubmesh;
677 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
678 boxSubmesh.StartIndexLocation = 0;
679 boxSubmesh.BaseVertexLocation = 0;
680
681
682 std::vector<Vertex> vertices(box.Vertices.size());
683
684 for (size_t i = 0; i < box.Vertices.size(); ++i)
685 {
686 vertices[i].Pos = box.Vertices[i].Position;
687 vertices[i].Normal = box.Vertices[i].Normal;
688 vertices[i].TexC = box.Vertices[i].TexC;
689 }
690
691 std::vector<std::uint16_t> indices = box.GetIndices16();
692
693 const UINT vbByteSize = (UINT)vertices.size() * sizeof(Vertex);
694 const UINT ibByteSize = (UINT)indices.size() * sizeof(std::uint16_t);
695
696 auto geo = std::make_unique<MeshGeometry>();
697 geo->Name = "boxGeo";
698
699 ThrowIfFailed(D3DCreateBlob(vbByteSize, &geo->VertexBufferCPU));
700 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
701
702 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
703 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
704
705 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
706 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
707
708 geo->IndexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
709 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);
710

907void Game::WaitForGPU()
908{
909 // Schedule a Signal command in the queue
910 ThrowIfFailed(mCommandQueue->Signal(mFence.Get(), mCurrentFence));
911
912 // Wait until the GPU has completed commands up to this fence point.
913 if (mFence->GetCompletedValue() < mCurrentFence)
914 {
915 HANDLE eventHandle = CreateEventEx(nullptr, FALSE, FALSE, EVENT_ALL_ACCESS);
916
917 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrentFence, eventHandle));
918 WaitForSingleObject(eventHandle, INFINITE);
919 CloseHandle(eventHandle);
920 }
921
922 mCurrentFence++;
923}
